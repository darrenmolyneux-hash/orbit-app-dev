SELECT 
  hp.id,
  hp.part_ref,
  pt.name AS part_type,
  pt.is_battery,
  hp.spec,
  hp.part_make,
  hp.part_model,
  hp.serial_number,
  hp.condition_grade,
  hp.status,
  hp.is_battery,
  hp.removed_by,
  hp.removed_at::date AS removed_date,
  hp.disposition_due,
  hp.removal_notes,
  hp.destination_type,
  hp.current_location_id,
  hp.installed_in_asset_id,
  a.asset_ref AS installed_in_ref,
  l.location_code AS current_location_code,
  l.location_name AS current_location_name,
  CASE 
    WHEN hp.disposition_due < CURRENT_DATE THEN true 
    ELSE false 
  END AS is_overdue
FROM harvested_parts hp
LEFT JOIN part_types pt ON pt.part_type_id = hp.part_type_id
LEFT JOIN assets a ON a.asset_id = hp.installed_in_asset_id
LEFT JOIN locations l ON l.location_id = hp.current_location_id
WHERE hp.source_asset_id = {{appsmith.URL.queryParams.asset_id}}
ORDER BY hp.removed_at DESC;
