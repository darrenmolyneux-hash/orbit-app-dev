SELECT 
  pt.name AS part_type,
  hp.part_cost,
  hp.serial_number,
  hp.condition_grade,
  hp.removed_at
FROM harvested_parts hp
LEFT JOIN pre_inventory_assessments pia 
  ON pia.harvested_part_id = hp.id
LEFT JOIN part_types pt 
  ON pt.part_type_id = pia.part_type_id
WHERE hp.installed_in_asset_id = {{Number(appsmith.URL.queryParams.asset_id)}}
ORDER BY hp.removed_at DESC;
