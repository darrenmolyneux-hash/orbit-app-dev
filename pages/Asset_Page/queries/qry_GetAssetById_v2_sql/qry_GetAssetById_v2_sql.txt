SELECT
  a.*,
  it.item_type_name,
  mk.make_name,
  md.model_name,
  st.status_name,
  st.category AS status_category
FROM assets a
LEFT JOIN item_types it ON it.item_type_id = a.item_type::integer
LEFT JOIN makes mk ON mk.make_id = a.make::integer
LEFT JOIN models md ON md.model_id = a.model::integer
LEFT JOIN status_types st ON st.status_id = a.status_id
WHERE a.asset_id = {{appsmith.URL.queryParams.asset_id}};
