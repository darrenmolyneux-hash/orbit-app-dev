SELECT
  hp.id               AS harvested_part_id,
  NULL::integer       AS donor_pia_id,
  pt.name             AS part_type,
  pt.is_battery,
  mk.make_name        AS donor_make,
  mo.model_name       AS donor_model,
  src.asset_ref       AS donor_asset_ref,
  src.asset_id        AS donor_asset_id,
  hp.serial_number,
  hp.condition_grade,
  hp.is_salvageable,
  hp.part_cost,
  'in_stock'          AS availability
FROM harvested_parts hp
LEFT JOIN pre_inventory_assessments pia_src
  ON pia_src.harvested_part_id = hp.id
LEFT JOIN part_types pt
  ON pt.part_type_id = pia_src.part_type_id
LEFT JOIN assets src
  ON src.asset_id = hp.source_asset_id
LEFT JOIN makes mk
  ON mk.make_id = src.make
LEFT JOIN models mo
  ON mo.model_id = src.model
WHERE hp.status = 'in_stock'
AND (
  COALESCE(mo.model_name, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
  OR COALESCE(mk.make_name, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
  OR COALESCE(src.asset_ref, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
  OR COALESCE(pt.name, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
)

UNION ALL

SELECT
  NULL::integer       AS harvested_part_id,
  pia.id              AS donor_pia_id,
  pt.name             AS part_type,
  pt.is_battery,
  mk.make_name        AS donor_make,
  mo.model_name       AS donor_model,
  src.asset_ref       AS donor_asset_ref,
  src.asset_id        AS donor_asset_id,
  NULL                AS serial_number,
  NULL                AS condition_grade,
  true                AS is_salvageable,
  NULL                AS part_cost,
  'in_donor_asset'    AS availability
FROM pre_inventory_assessments pia
JOIN part_types pt
  ON pt.part_type_id = pia.part_type_id
JOIN assets src
  ON src.asset_id = pia.asset_id
JOIN makes mk
  ON mk.make_id = src.make
JOIN models mo
  ON mo.model_id = src.model
WHERE pia.harvested_part_id IS NULL
AND pia.is_salvageable = true
AND pia.asset_id != {{Number(appsmith.URL.queryParams.asset_id)}}
AND (
  COALESCE(mo.model_name, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
  OR COALESCE(mk.make_name, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
  OR COALESCE(src.asset_ref, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
  OR COALESCE(pt.name, '') ILIKE '%' || COALESCE({{inp_part_search.text}}, '') || '%'
)
ORDER BY donor_make, donor_model, part_type;
