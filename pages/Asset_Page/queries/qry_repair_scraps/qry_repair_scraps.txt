SELECT 
  hp.id,
  hp.part_ref,
  pt.name AS part_type,
  hp.removal_notes,
  hp.removed_by,
  hp.removed_at::date AS removed_date,
  hp.destination_type
FROM harvested_parts hp
LEFT JOIN part_types pt ON pt.part_type_id = hp.part_type_id
WHERE hp.source_asset_id = {{Number(appsmith.URL.queryParams.asset_id)}}
AND hp.status = 'scrapped'
ORDER BY hp.removed_at DESC;
