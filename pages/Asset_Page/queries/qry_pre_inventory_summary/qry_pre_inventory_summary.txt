SELECT
  COUNT(*) AS total,
  COUNT(*) FILTER (WHERE is_salvageable = true) AS salvageable,
  COUNT(*) FILTER (WHERE is_salvageable = false) AS scrapped,
  COUNT(*) FILTER (WHERE harvested_part_id IS NOT NULL) AS removed,
  COUNT(*) FILTER (WHERE is_salvageable = true AND harvested_part_id IS NULL) AS still_in_asset,
  0 AS overdue,
  0 AS due_soon,
  MIN(assessed_at) AS assessed_at,
  MAX(assessed_by) AS assessed_by
FROM pre_inventory_assessments
WHERE asset_id = {{Number(appsmith.URL.queryParams.asset_id)}};
