SELECT
  a.*,
  it.item_type_name,
  mk.make_name,
  mo.model_name,
  cu.customer_name
FROM assets a
LEFT JOIN item_types it ON it.item_type_id = a.item_type
LEFT JOIN makes mk ON mk.make_id = a.make
LEFT JOIN models mo ON mo.model_id = a.model
LEFT JOIN collections c ON c.collection_id = a.collection_id
LEFT JOIN customers cu ON cu.customer_id = c.customer_id
WHERE a.asset_id = {{Number(appsmith.URL.queryParams.asset_id)}};
