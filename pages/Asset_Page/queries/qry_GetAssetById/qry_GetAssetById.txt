SELECT
  a.*,
  it.item_type_name,
  cu.customer_name
FROM assets a
LEFT JOIN item_types it ON it.item_type_name = a.item_type
LEFT JOIN collections c ON c.collection_id = a.collection_id
LEFT JOIN customers cu ON cu.customer_id = c.customer_id
WHERE a.asset_id = {{appsmith.URL.queryParams.asset_id}};
