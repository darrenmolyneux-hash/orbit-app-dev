SELECT a.grade,
       COUNT(*) AS units,
       ROUND(AVG(a.original_cost), 2) AS avg_paid,
       MIN(a.original_cost) AS min_paid,
       MAX(a.original_cost) AS max_paid
FROM assets a
WHERE a.make = {{ qry_GetAssetById.data[0].make }}
  AND a.model = {{ qry_GetAssetById.data[0].model }}
  AND a.original_cost IS NOT NULL
  AND a.date_added >= NOW() - INTERVAL '6 months'
GROUP BY a.grade
ORDER BY a.grade;
