INSERT INTO pre_inventory_assessments (
  asset_id,
  part_type_id,
  is_salvageable,
  assessed_by,
  assessed_at,
  harvested_part_id,
  notes
)
SELECT
  {{Number(appsmith.URL.queryParams.asset_id)}},
  pia_src.part_type_id,
  hp.is_salvageable,
  {{appsmith.user.name}},
  NOW(),
  hp.id,
  'Installed from stock'
FROM harvested_parts hp
LEFT JOIN pre_inventory_assessments pia_src 
  ON pia_src.harvested_part_id = hp.id
WHERE hp.id = {{Number(appsmith.store.hp_install_id)}};
