INSERT INTO pre_inventory_assessments (
  asset_id,
  part_type_id,
  is_salvageable,
  assessed_by,
  assessed_at,
  harvested_part_id,
  notes
)
SELECT
  {{Number(appsmith.URL.queryParams.asset_id)}},
  hp.part_type_id,
  hp.is_salvageable,
  {{appsmith.store.userName}},
  NOW(),
  hp.id,
  'Installed from stock'
FROM harvested_parts hp
WHERE hp.id = {{Number(appsmith.store.hp_install_id)}}
  AND hp.part_type_id IS NOT NULL
ON CONFLICT (asset_id, part_type_id)
DO UPDATE SET
  harvested_part_id = EXCLUDED.harvested_part_id,
  is_salvageable    = EXCLUDED.is_salvageable,
  assessed_by       = EXCLUDED.assessed_by,
  assessed_at       = EXCLUDED.assessed_at,
  notes             = EXCLUDED.notes;
