-- qry_set_new_password (reused later for admin reset + self-service change)
UPDATE users
SET password_hash = crypt({{this.params.newPassword}}, gen_salt('bf')),
    must_change_password = {{this.params.mustChange}},
    failed_login_attempts = 0,
    locked_until = NULL
WHERE user_id = {{this.params.userId}};
