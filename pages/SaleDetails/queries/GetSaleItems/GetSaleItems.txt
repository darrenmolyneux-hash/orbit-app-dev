SELECT 
    si.sell_price,
    a.asset_ref,
    a.serial_number,
    mk.make_name,
    mo.model_name
FROM sale_items si
JOIN assets a ON si.asset_id = a.asset_id
LEFT JOIN makes mk ON mk.make_id = a.make
LEFT JOIN models mo ON mo.model_id = a.model
WHERE si.sale_id = {{Number(appsmith.URL.queryParams.sale_id)}};
