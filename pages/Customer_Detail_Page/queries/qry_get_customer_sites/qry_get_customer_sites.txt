SELECT
  s.site_id,
  s.address1,
  s.address2,
  s.city,
  s.postcode,
  s.site_contact,
  s.contact_number,
  s.contact_email,
  s.created_at::date AS created_date,
  ss.survey_id,
  ss.survey_date,
  ss.last_reviewed_date,
  ss.surveyed_by,
  CASE
    WHEN ss.survey_id IS NULL THEN 'no_survey'
    WHEN ss.last_reviewed_date < CURRENT_DATE - INTERVAL '12 months' THEN 'expired'
    WHEN ss.last_reviewed_date < CURRENT_DATE - INTERVAL '11 months' THEN 'due_soon'
    ELSE 'current'
  END AS survey_status
FROM sites s
LEFT JOIN site_surveys ss ON ss.site_id = s.site_id
WHERE s.customer_id = {{Number(appsmith.URL.queryParams.customer_id)}}
ORDER BY s.created_at DESC;
