SELECT
  c.collection_id,
  c.collection_date,
  c.status,
  c.address1,
  COUNT(a.asset_id) AS asset_count
FROM collections c
LEFT JOIN assets a ON a.collection_id = c.collection_id
WHERE c.customer_id = {{Number(appsmith.URL.queryParams.customer_id)}}
GROUP BY c.collection_id, c.collection_date, c.status, c.address1
ORDER BY c.collection_date DESC;
