SELECT COUNT(*) AS total
FROM assets a
LEFT JOIN item_types it ON it.item_type_id = a.item_type
LEFT JOIN makes mk ON mk.make_id = a.make
LEFT JOIN models md ON md.model_id = a.model
LEFT JOIN status_types st ON st.status_id = a.status_id
WHERE (
  {{appsmith.URL.queryParams.collectionId ? Number(appsmith.URL.queryParams.collectionId) : 0}} = 0
  OR a.collection_id = {{appsmith.URL.queryParams.collectionId ? Number(appsmith.URL.queryParams.collectionId) : 0}}
)
AND (
  {{appsmith.store.asset_status_filter || ''}} = ''
  OR st.status_name = {{appsmith.store.asset_status_filter || ''}}
)
AND (
  {{appsmith.store.asset_search || ''}} = ''
  OR a.asset_ref ILIKE {{'%' + (appsmith.store.asset_search || '') + '%'}}
  OR a.serial_number ILIKE {{'%' + (appsmith.store.asset_search || '') + '%'}}
);
