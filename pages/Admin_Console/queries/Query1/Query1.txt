SELECT
  CASE
    WHEN s.sale_date::date = CURRENT_DATE THEN 'today'
    WHEN s.sale_date::date >= CURRENT_DATE - INTERVAL '7 days' THEN '7days'
    WHEN s.sale_date::date >= CURRENT_DATE - INTERVAL '30 days' THEN '30days'
    WHEN s.sale_date::date >= DATE_TRUNC('month', CURRENT_DATE) THEN 'mtd'
  END AS period,
  a.sell_price
FROM sale_items si
JOIN sales s ON s.sale_id = si.sale_id
JOIN assets a ON a.asset_id = si.asset_id
WHERE s.sale_date >= CURRENT_DATE - INTERVAL '30 days'
ORDER BY period;
