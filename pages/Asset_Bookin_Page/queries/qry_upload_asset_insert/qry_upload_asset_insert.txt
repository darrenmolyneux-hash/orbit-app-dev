INSERT INTO assets (
  collection_id,
  customer,
  item_type,
  make,
  model,
  serial_number,
  asset_ref,
  data_bearing,
  added_by,
  date_added,
  status_id
)
SELECT
  {{Number(appsmith.URL.queryParams.collectionId)}},
  (SELECT cu.customer_id::integer FROM customers cu
   JOIN collections c ON c.customer_id = cu.customer_id
   WHERE c.collection_id = {{Number(appsmith.URL.queryParams.collectionId)}}),
  {{Number(appsmith.store.pendingItemType)}},
  {{Number(appsmith.store.pendingMake)}},
  {{Number(appsmith.store.pendingModel)}},
  '{{appsmith.store.pendingSerial}}',
  '{{this.params.assetRef}}',
  {{appsmith.store.pendingDataBearing === true}},
  '{{appsmith.user.name}}',
  NOW(),
  (SELECT status_id FROM status_types WHERE status_name = 'Received');
