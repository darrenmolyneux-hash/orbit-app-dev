WITH ref_prefix AS (
    SELECT next_ref FROM collection_asset_refs WHERE collection_id = {{appsmith.URL.queryParams.collectionId}}
),
seq AS (
    UPDATE global_asset_counter
    SET next_number = next_number + 1
    WHERE id = 1
    RETURNING next_number - 1 AS num
)
SELECT 
    ref_prefix.next_ref || '-' || LPAD(seq.num::text, 5, '0') AS asset_ref
FROM ref_prefix, seq;
