UPDATE global_asset_counter
SET next_number = next_number + 1
WHERE id = 1
RETURNING 
    LPAD(
        (SELECT next_ref FROM collection_asset_refs 
         WHERE collection_id = {{Number(appsmith.URL.queryParams.collectionId)}})::text, 
        3, '0'
    )
    || '-' ||
    LPAD((next_number - 1)::text, 5, '0') AS asset_ref;
