SELECT date_trunc('month', s.sale_date) AS month,
       a.grade,
       ROUND(AVG(si.sell_price), 2) AS avg_sold,
       COUNT(*) AS units_sold
FROM assets a
JOIN sale_items si ON si.asset_id = a.asset_id
JOIN sales s ON s.sale_id = si.sale_id
WHERE a.model = {{ appsmith.store.selectedModel }}
  AND s.sale_date >= NOW() - INTERVAL '6 months'
  AND a.grade IS NOT NULL
GROUP BY month, a.grade
ORDER BY month, a.grade;
