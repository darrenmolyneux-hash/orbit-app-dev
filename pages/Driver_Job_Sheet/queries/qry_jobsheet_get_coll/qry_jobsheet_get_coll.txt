SELECT c.*, cu.customer_name,
  ss.vehicle_access_height, ss.vehicle_access_height_notes,
  ss.parking_loading_bay, ss.parking_notes,
  ss.lift_restrictions, ss.lift_notes,
  ss.narrow_corridors, ss.corridor_notes,
  ss.stairs_involved, ss.stairs_notes,
  ss.escort_required,
  ss.site_induction_required,
  ss.ppe_requirements,
  ss.site_hazards,
  ss.manual_handling_required, ss.manual_handling_notes,
  ss.security_procedures, ss.security_details,
  ss.driver_instructions,
  ss.surveyed_by, ss.survey_date, ss.last_reviewed_date
FROM collections c
LEFT JOIN customers cu ON cu.customer_id = c.customer_id
LEFT JOIN site_surveys ss ON ss.site_id = c.site_id
WHERE c.collection_id = {{this.params.collectionId}};
