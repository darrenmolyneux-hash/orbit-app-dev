SELECT 
  c.collection_id,
  c.collection_date,
  c.status,
  c.address1,
  c.city,
  cu.customer_name
FROM collections c
LEFT JOIN customers cu ON cu.customer_id = c.customer_id
WHERE c.status('Booked')
AND (
  {{inp_driver_search.text}} = ''
  OR cu.customer_name ILIKE '%' || {{inp_driver_search.text}} || '%'
  OR CAST(c.collection_id AS text) ILIKE '%' || {{inp_driver_search.text}} || '%'
)
ORDER BY c.collection_date ASC;
