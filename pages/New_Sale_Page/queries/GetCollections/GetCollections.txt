SELECT 
    c.collection_id,
    c.customer_id,
    cu.customer_name,
    c.site_id,
    c.address1,
    c.address2,
    c.city,
    c.postcode,
    c.collection_date,
    c.job_type,
    c.created_by,
    c.instructions,
    c.notes,
    c.status
FROM collections c
LEFT JOIN customers cu ON cu.customer_id = c.customer_id
WHERE c.status = 'Booked'
ORDER BY c.collection_id DESC;
