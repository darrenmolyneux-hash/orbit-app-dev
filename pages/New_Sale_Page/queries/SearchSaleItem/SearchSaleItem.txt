SELECT 
  a.*,
  st.status_name,
  it.item_type_name,
  mk.make_name,
  mo.model_name
FROM assets a
LEFT JOIN status_types st ON st.status_id = a.status_id
LEFT JOIN item_types it ON it.item_type_id::text = a.item_type
LEFT JOIN makes mk ON mk.make_id::text = a.make
LEFT JOIN models mo ON mo.model_id::text = a.model
WHERE 
  a.asset_ref = {{AssetRefSearch.text}}
  OR a.serial_number = {{SerialSearch.text}}
