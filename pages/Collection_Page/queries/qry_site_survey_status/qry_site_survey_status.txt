SELECT
  ss.survey_id,
  ss.surveyed_by,
  ss.survey_date,
  ss.last_reviewed_date,
  ss.vehicle_access_height,
  ss.parking_loading_bay,
  ss.lift_restrictions,
  ss.narrow_corridors,
  ss.stairs_involved,
  ss.escort_required,
  ss.site_induction_required,
  ss.ppe_requirements,
  ss.site_hazards,
  ss.manual_handling_required,
  ss.security_procedures,
  ss.security_details,
  ss.driver_instructions,
  CASE
    WHEN ss.survey_id IS NULL THEN 'no_survey'
    WHEN ss.last_reviewed_date < CURRENT_DATE - INTERVAL '12 months' THEN 'expired'
    WHEN ss.last_reviewed_date < CURRENT_DATE - INTERVAL '11 months' THEN 'due_soon'
    ELSE 'current'
  END AS survey_status
FROM sites s
LEFT JOIN site_surveys ss ON ss.site_id = s.site_id
WHERE s.site_id = {{Number(appsmith.store.col_site_id)}}
ORDER BY ss.survey_id DESC
LIMIT 1;
