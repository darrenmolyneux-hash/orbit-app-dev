-- qry_pre_inventory_insert — SIMPLIFIED (with notes kept)
-- Pre-inventory records the salvageable/scrap decision plus an
-- optional quick reason note.

INSERT INTO pre_inventory_assessments (
  asset_id,
  part_type_id,
  is_salvageable,
  notes,
  assessed_by,
  assessed_at
) VALUES (
  {{Number(appsmith.URL.queryParams.asset_id)}},
  {{Number(appsmith.store.pi_part_type_id)}},
  {{appsmith.store.pi_salvageable}},
  '{{appsmith.store.pi_notes}}',
  '{{appsmith.store.userName}}',
  NOW()
)
ON CONFLICT (asset_id, part_type_id)
DO UPDATE SET
  is_salvageable   = EXCLUDED.is_salvageable,
  notes            = EXCLUDED.notes,
  assessed_by      = EXCLUDED.assessed_by,
  assessed_at      = EXCLUDED.assessed_at
RETURNING id;
