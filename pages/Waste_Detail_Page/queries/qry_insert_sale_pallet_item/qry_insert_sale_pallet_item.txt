INSERT INTO pallet_items (pallet_id, item_type, asset_id, sale_price, notes, added_by, pallet_type)
SELECT
  {{this.params.palletId}},
  'asset',
  {{this.params.assetId}},
  {{this.params.salePrice}},
  '{{this.params.notes.split("'").join("''")}}',
  '{{this.params.addedBy}}',
  'sale'
WHERE NOT EXISTS (
  SELECT 1 FROM pallet_items
  WHERE pallet_id = {{this.params.palletId}}
  AND asset_id = {{this.params.assetId}}
  AND pallet_type = 'sale'
)
RETURNING item_id;
