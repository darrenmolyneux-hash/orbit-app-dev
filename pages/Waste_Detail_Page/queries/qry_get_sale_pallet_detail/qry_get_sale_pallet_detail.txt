SELECT
  sp.pallet_id,
  sp.pallet_ref,
  sp.status,
  sp.notes,
  sp.created_by,
  sp.created_at::date AS created_date,
  sp.closed_at::date AS closed_date,
  sp.shipped_at::date AS shipped_date,
  sp.customer_id,
  c.customer_name,
  COALESCE(SUM(pi.sale_price), 0) AS total_value,
  COUNT(pi.item_id) AS item_count
FROM sale_pallets sp
LEFT JOIN customers c ON c.customer_id = sp.customer_id
LEFT JOIN pallet_items pi ON pi.pallet_id = sp.pallet_id AND pi.pallet_type = 'sale'
WHERE sp.pallet_id = {{ Number(appsmith.URL.queryParams.pallet_id) }}
GROUP BY sp.pallet_id, sp.pallet_ref, sp.status, sp.notes, sp.created_by,
  sp.created_at, sp.closed_at, sp.shipped_at, sp.customer_id, c.customer_name;
