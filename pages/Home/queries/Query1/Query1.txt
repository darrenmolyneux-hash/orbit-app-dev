SELECT 
  'asset' AS result_type,
  a.asset_id::text AS id,
  a.asset_ref AS ref,
  a.serial_number AS detail,
  it.item_type_name AS label,
  mk.make_name AS make,
  mo.model_name AS model,
  NULL AS customer_name
FROM assets a
LEFT JOIN item_types it ON it.item_type_id = a.item_type
LEFT JOIN makes mk ON mk.make_id = a.make
LEFT JOIN models mo ON mo.model_id = a.model
WHERE 
  a.asset_ref ILIKE {{'%' + inp_global_search.text + '%'}}
  OR a.serial_number ILIKE {{'%' + inp_global_search.text + '%'}}
  OR mk.make_name ILIKE {{'%' + inp_global_search.text + '%'}}
  OR mo.model_name ILIKE {{'%' + inp_global_search.text + '%'}}

UNION ALL

SELECT 
  'collection' AS result_type,
  c.collection_id::text AS id,
  c.collection_id::text AS ref,
  cu.customer_name AS detail,
  'Collection' AS label,
  NULL AS make,
  NULL AS model,
  cu.customer_name AS customer_name
FROM collections c
JOIN customers cu ON cu.customer_id = c.customer_id
WHERE 
  c.collection_id::text ILIKE {{'%' + inp_global_search.text + '%'}}
  OR cu.customer_name ILIKE {{'%' + inp_global_search.text + '%'}}

ORDER BY result_type, ref
LIMIT 20;
