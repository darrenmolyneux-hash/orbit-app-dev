SELECT
  COUNT(*) FILTER (WHERE DATE_TRUNC('month', collection_date) = DATE_TRUNC('month', CURRENT_DATE)) AS this_month,
  COUNT(*) FILTER (WHERE DATE_TRUNC('month', collection_date) = DATE_TRUNC('month', CURRENT_DATE - INTERVAL '1 month')) AS last_month,
  COUNT(*) FILTER (WHERE DATE_TRUNC('year', collection_date) = DATE_TRUNC('year', CURRENT_DATE)) AS this_year,
  COUNT(*) FILTER (WHERE DATE_TRUNC('year', collection_date) = DATE_TRUNC('year', CURRENT_DATE - INTERVAL '1 year')) AS last_year
FROM collections;
